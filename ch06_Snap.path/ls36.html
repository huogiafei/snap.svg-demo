Snap.path.intersection()

/**
 * Snap.path.intersection(path1,path2)
 * return:
 * x,y(相交点)
 * t1,t2(片段:0~1)
 * segment1,segment2 (路径片断的顺序号)
 * bez1,bez2 (path1和path2各自的贝塞尔曲线坐标)
 */
var p = Snap('#paper');
var line1Path = 'M10 10 L90 90';
var line2Path = 'M10 90 L90 10';
var line1 = p.path(line1Path);
var line2 = p.path(line2Path);
p.g(line1, line2).attr({
    strokeWidth: 4,
    stroke: '#087',
    strokeOpacity: '.8'
});
var result = Snap.path.intersection(line1Path, line2Path);
console.log(result);
var html = "";
for (var key in result[0]) {
    html += `<p>${key}: ${result[0][key]}</p>`
}
document.querySelector('#console').innerHTML += html;
