getBBox , getPointAtLength , getSubpath , getTotalLength

Snap.path.toCubic()

var p = Snap("#paper");
var linePath = 'M10 10 L90 90';
var line = p.path(linePath).attr({
    strokeWidth:3,
    stroke:'#fc1'
})
function GetBBox(){
    var result = Snap.path.getBBox(line);
    console.log(result);
}
function GetPointAtLength(){
    /*a half of line*/
    var result = Snap.path.getPointAtLength(linePath,56.5685);
    console.log(result);
}
function GetSubpath(){
    /*(20,20) to (70,70)*/
    var result = Snap.path.getSubpath(linePath,14.1421,99);
}
function GetTotalLength(){
    var result = Snap.path.getTotalLength(linePath);
}
