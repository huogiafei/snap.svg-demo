Set.push() / pop() / forEach()

window.onload = function () {
    for (var i = 0; i < 5; i++) {
        p.circle(80 + 100 * i, 50, 40).attr({fill: '#087'});
    }
    ballSets = p.selectAll('circle');
};
function step1() {
    ballSets.pop().attr({
        fill: 'yellowgreen'
    })
}
function step2() {
    p2.append(ballSets);
}
function step3() {
    ballSets.forEach(function (item) {
        item.attr({fill: 'yellow'})
    })
}
