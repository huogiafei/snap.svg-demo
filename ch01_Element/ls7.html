remove , select

Element.remove()

/*Removes element from the DOM*/
var p = Snap('#paper');
var parent = p.circle(50,50,45).attr({
    stroke:'#fff',
    strokeWidth:2,
    fill:'transparent'
});
var circle = p.circle(50,50,35).attr({
    fill:'#087'
})
var rect = p.rect(40,40,20,20).attr({
    fill:'#2ECC71'
})
function removeCircle(){
    circle.remove();
}
function removeRect(){
    rect.remove();
}
function reset(){
    p.append(circle);
    p.append(rect);
}

Element.select()/selectAll()

/*Gathers the nested Element matching the given set of CSS selectors*/
var p2 = Snap('#paper2');
var c1 = p2.circle(50,50,40);
var c2 = p2.circle(200,50,40);
var r1 = p2.rect(100,48,50,4);
function selectRect() {
    var rect = p2.select('rect');
    rect.attr({fill: 'yellow'})
}
function selectAllCircles() {
    var circles = p2.selectAll('circle');
    circles.forEach(function(c){
        c.attr({fill:'#fc1'})
    });
}
