Transform , Animate

Element.transform()

window.onload = function () {
    var p = Snap('#paper');
    var rect = p.rect(20,20,60,60,5,5).attr({
        fill:'orange',
        fillOpacity:'0.4'
    }).transform('translate(10,10)');
    console.dir(rect.transform());
    $$("#console")[0].innerHTML = `<p>circle.transform(): ${rect.transform()}</p>`;
    $$("#transBtn")[0].addEventListener('click',trans);
    function trans(){
        rect.transform('rotate(45,50,50)');
        $$("#console")[0].innerHTML += `<p>transform2: ${rect.transform()}</p>`;
    }
}

Element.animate() / inAnim() / stop()

var p2 = Snap("#paper2");
var c = p2.circle(50, 50, 40).addClass('borderCircle').attr({
    color: "#f66",
    fillOpacity: 0.1
});
function go() {
    c.animate({
        cx: 400
    }, 6000, mina.elastic, function () {
        this.animate({
            fillOpacity: 0.6
        }, 1000, mina.easeout);
    })
    setTimeout(function(){
        console.dir(c.inAnim());
    }, 7000);
}
function stop(){
    c.stop();
}
