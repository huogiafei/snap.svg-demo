var p = Snap('#paper');
var c = p.circle(50,50,40);
circleNode = c.node;
circleNode.onclick = function(e){
    var cons = document.querySelector('#console')
    cons.textContent = 'Hello Snap.svg'
}

Element.type()

var p = Snap('#paper2');
var shape1 = p.rect(0, 0, 10, 20).attr({
    fill: '#f66',
});
var shape2 = p.circle(20, 50, 10).attr({
    fill: '#f66',
});
var cons = document.querySelector('#console2')
cons.textContent = shape1.type + '/' + shape2.type;
