Element.data() / removeData()

var p = Snap('#paper');
var pointArray = [50, 10, 86, 40, 70, 80, 30, 80, 14, 40];
var star = p.polygon(pointArray);
star.attr({
    stroke: '#fc1',
    strokeWidth: 3,
    fill: 'none'
})
function setData() {
    star.data('point', pointArray);
    //Don’t confuse with data- attributes
    star.node.setAttribute('point', '123');
}
function removeData() {
    //If key is not provided, removes all the data of the element.
    star.removeData();
}
