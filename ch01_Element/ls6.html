before , after

Element.before() / after()

var p = Snap('#paper');
var e1 = p.ellipse(30, 50, 30, 40).attr({
    fill: 'yellow',
    fillOpacity:'0.8'
});
var e2 = p.ellipse(60, 50, 30, 40).attr({
    fill: 'orange',
    fillOpacity:'0.8'
});
function Before() { //inserts e2 before e1
    e1.before(e2);
}
function After() { //inserts e2 after e1
    e1.after(e2);
}
function InsertAfter() { //inserts e1 after e2
    e1.insertAfter(e2);
}
function InsertBefore() { //inserts e2 after e1
    e1.insertBefore(e2);
}
