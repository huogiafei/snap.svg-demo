append , prepend

Element.append() / prepend()

var p = Snap('#paper');
var p2 = Snap('#paper2');
p.attr({strokeWidth:3});

var circle = p.circle(50,50,40).attr({fill:'#fc1'});
var polygon = p.polygon(30,30, 50,70, 70,30).attr('fill','#111');
var rect = p2.rect(10,10,80,80).attr({fill:'#087'});
var line = p2.line(10,30,90,70).attr({stroke:'#fff'});

p2.append(polygon);
line.appendTo(p);
p2.prepend(circle);
rect.prependTo(p);
