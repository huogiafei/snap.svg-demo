Mouse

Element.mousedown() / mouseup()

var p = Snap('#paper');
var c1 = p.circle(50, 50, 40).attr({
    fill: '#f87',
    fillOpacity: 1,
    cursor: 'pointer',
    strokeWidth: 3
});
var CurObj;
c1.mousedown(MouseDown).mouseup(MouseUp);
function MouseDown() {
    var obj = this;
    var opacity = +obj.attr('fillOpacity') - 0.1;
    var newCircle = obj.clone().attr({
        cx: '+=80',
        fillOpacity: opacity
    });
    CurObj = this;
    obj.unmousedown(MouseDown).attr({cursor: 'default'});
    newCircle.mousedown(MouseDown).mouseup(MouseUp);
}
function MouseUp() {
    CurObj.attr({
        fill: 'none',
        stroke: '#f87'
    })
}

Element.mousemove() / mouseout()

var p = Snap('#paper2');
var c2 = p.circle(50, 50, 40).attr({
    color:'#fd1',
}).addClass('border-shape');
c2.mousemove(MouseMove);
c2.mouseout(MouseOut);
function MouseMove(){
    var randomColor = "#" + parseInt(Math.floor(Math.random()*999)+1);
    this.attr({
        color:randomColor
    });
}
function MouseOut(){
    this.attr({
        color:'#000'
    })
}

Element.mouseover()

var p = Snap('#paper3');
var c3 = p.circle(50, 50, 40).attr({
    color: '#0c9',
}).addClass('border-shape');
c3.mouseover(MouseOver);
function MouseOver() {
    this.attr({
        cx: '+=100'
    });
}
