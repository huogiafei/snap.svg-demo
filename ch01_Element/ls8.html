Class

Element.addClass() / removeClass() / toggleClass() / hasClass()

var p = Snap('#paper');
var rect = p.rect(10,10,80,80);
function AddClass(){
    rect.addClass('style-1')
}
function RemoveClass(){
    rect.removeClass('style-1')
}
function ToggleClass(){
    //toggleClass(string,true) = addClass
    //toggleClass(string,false) = removeClass
    rect.toggleClass('style-1');
}
