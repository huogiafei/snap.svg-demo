getPointAtLength , getSubpath , getTotalLength

Element.getPointAtLength()

/* Returns the coordinates of the point located
   at the given length along the given path */
//<line> element getPointAtLength is disabled
var line = p.path('m0,0 L 60 100').attr({
    fill: 'none',
    stroke: '#087',
    strokeOpacity: 0.9,
    strokeWidth: 3,
var pointObj = line.getPointAtLength(60);
console.dir(pointObj); // a half
var point = p.circle(pointObj.x, pointObj.y, 3).attr({
    fill: '#e8532e',
})

Element.getSubpath()

/* Returns the subpath of a given path between given start and end lengths */
var p2 = Snap('#paper2');
var line2 = p2.path("m20,20 l80 80").attr({
    fill: 'none',
    strokeWidth: 3,
    stroke: '#f85'
});

var subPath = line2.getSubpath(28, 72);
console.log(subPath);
var subLine = p2.path(subPath).attr({
    fill: 'none',
    strokeWidth: 3,
    stroke: '#087'
})

Element.getTotalLength()

/* Returns the length of the path in pixels (only works for path elements) */
var p3 = Snap('#paper3');
var arc = p3.path('M50,10 l 40 40 l -40 40 l -40 -40 z');
arc.attr({
    fill:'none',
    stroke:'#09e',
    strokeWidth:3
})
var totalLen = arc.getTotalLength();
console.log(totalLen);//40*sqrt(2) *4
