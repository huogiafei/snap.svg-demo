Element.use()

var p = Snap('#paper');
var rect = p.rect(20,20,60,60).attr({
    fillOpacity:0.4
});
function Use(){
    rect.use().transform('rotate(30,50,50)');
    rect.use().transform('rotate(-30,50,50)');
    document.querySelector("#console").innerHTML = htmlToTXT(p.outerSVG());
}

Element.toPattern()

var p2 = Snap('#paper2');
var tile = p2.path("m10,0 l10 10 l-10 10 l-10 -10 z")
        .attr('fill', 'hsla(0,0%,100%,0.4').toPattern(0, 0, 20, 20);
var rect2 = p2.rect(0, 0, 200, 100).attr({fill: tile});
var rect3 = p2.rect(0, 0, 200, 100).attr({fill: '#087'});
rect3.after(rect2);

Element.marker()

var p3 = Snap('#paper3');
p3.attr({color:'#fc1'});
var startMarker = p3.circle(3, 3, 2, 2).addClass('arrow-line');
var endMarker = p3.path('m5,1 l4 4 l-4 4').addClass('arrow-line');
var s_m = startMarker.marker(0, 0, 6, 6, 6, 3);
var e_m = endMarker.marker(0, 0, 10, 10, 7, 5);
var line = p3.line(20, 50, 100, 50).attr({
    markerStart: s_m,
    markerEnd: e_m,
    strokeWidth:3
}).addClass('arrow-line');
