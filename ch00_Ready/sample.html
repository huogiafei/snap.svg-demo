var p = Snap("#paper");
var line = p.line(50, 50, 750, 50);
var point = p.circle(5, 5, 2.5, 2.5).attr({
    fill: '#fff'
});
var marker = point.marker(0, 0, 10, 10, 5, 5);
line.attr({
    stroke: '#eee',
    strokeWidth: 2,
    strokeDasharray: 10,
    markerStart: marker,
    markerEnd: marker
})
var c1 = p.circle(50, 50, 40);
c1.attr({
    fill: '#f66',
    fillOpacity: .8
});
c1.animate({
    cx: 750
}, 10000);
