filter3:saturate,sepia,shadow

Snap.filter.saturate() / sepia() / shadow

var p = Snap('#paper');
var img = p.image('../img/seed.png', 3, 3, 110, 120);
function Saturate(arg) {
    var f = p.filter(Snap.filter.saturate(arg));
    img.attr({filter: f})
}
/*sepia(0~1)*/
function Sepia(arg) {
    var f = p.filter(Snap.filter.sepia(arg));
    img.attr({filter: f})
}
/*shadow(x,y,blurDistance,color)*/
function Shadow(x,y,w,c) {
    var f = p.filter(Snap.filter.shadow(x,y,w,c));
    img.attr({filter: f})
}
