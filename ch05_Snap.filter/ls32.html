filter2:grayscale,hueRotate,invert

Snap.filter.grayscale() / hueRotate() / invert

var p = Snap('#paper');
var img = p.image('../img/outdoors.jpg', 0, 0, 400, 300);
function Gray(arg) {
    var f = p.filter(Snap.filter.grayscale(arg));
    img.attr({filter: f})
}
/*HueRotate(0~360)*/
function HueRotate(arg) {
    var f = p.filter(Snap.filter.hueRotate(arg));
    img.attr({filter: f})
}
/*invert(0~1)*/
function Invert(arg) {
    var f = p.filter(Snap.filter.invert(arg));
    img.attr({filter: f})
}
