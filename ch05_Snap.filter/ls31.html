filter1:blur,brightness,contrast

Snap.filter.blur() / brightness() / contrast

var p = Snap('#paper');
var img = p.image('../img/gloom.png', 0, 0, 250, 200);
function Blur() {
    var f = p.filter(Snap.filter.blur(5));
    img.attr({filter: f})
}
/*brightness(0~1)*/
function Bright(arg) {
    var f = p.filter(Snap.filter.brightness(arg));
    img.attr({filter: f})
}
/*contrast(0~1)*/
function Contrast(arg) {
    var f = p.filter(Snap.filter.contrast(arg));
    img.attr({filter: f})
}
function reset() {
    img.attr({filter: 'none'})
}
