add,invert,clone

Matrix.add() / invert() / clone()

/* Matrix.add() : Adds the given matrix to existing one */
/* Matrix.invert() : Returns an inverted version of the matrix */
/* Matrix.clone() : Returns a copy of the matrix */
var p = Snap('#paper');
var circle = p.circle(150,50,40).attr({
    color:'orange'
}).addClass('border-shape');
var m1 = new Snap.Matrix(1,0,0,1,100,0);
circle.transform(m1.toTransformString());
function back(){
    m1.add(1,0,0,1,-200,0); //E: 100 - 200 = -100
    circle.transform(m1.toTransformString());
    console.log(m1);
    $$("#console")[0].innerHTML += `<p>m1 now:   ${m1}</p>`
}
function forward(){
    var m_i = m1.invert();
    console.log(m_i);
    $$("#console")[0].innerHTML += `<p>m_i(m1.invert): ${m_i}</p>`
}
function matrixClone(){
    var m2 = m1.clone();
    console.log(m2);
    $$("#console")[0].innerHTML += `<p>m2(m1.clone): ${m2}</p>`
}
