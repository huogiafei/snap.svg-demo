gradient , filter

Paper.gradient()

var p = Snap('#paper');
var lg = p.gradient('l(0,0,1,1)#087-#eee-#087');
var lg2 = p.gradient('l(0,0,1,0)#fff-#F39C12:30-#fc1');
var rg = p.gradient('r(0.5,0.5,0.5)#eee-#09e');
var rg2 = p.gradient('r(0.3,0.3,0.3)#eee-#f66');
var rect = p.rect(0,0,100,100).attr({fill:lg});
rect.clone().attr({
    x:110,
    fill:lg2
});
var circle = p.circle(260,50,40).attr({fill:rg});
circle.clone().attr({
    cx:350,
    fill:rg2
})

Paper.filter()

var p2 = Snap('#paper2');
var filter = p2.paper.filter('<feGaussianBlur stdDeviation="4"></feGaussianBlur>');
var gradient = p2.gradient('r(0.3,0.3,0.3)#eee-#f66');
p2.circle(50, 50, 40).attr({
    fill: filter,
}).clone().attr({
    fill: gradient,
    cx:'-=1',
    cy:'-=2'
})
