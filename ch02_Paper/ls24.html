/* Returns SVG code for the Paper */
var p = Snap('#paper');
var rect = p.rect(10,10,80,80,5,5).attr({fill:'#087'});
function svgToString(){
    console.log(rect.toString());
    $$('#console')[0].innerHTML = htmlToTXT(rect.toString());
}

/* Removes all child nodes of the paper, except <defs>. */
var p2 = Snap('#paper2');
window.onload = function(){
    var el1 = p2.rect(10,10,80,80,5,5).attr({fill:'orange'});
    var el2 = p2.rect(110,10,80,80,5,5).attr({fill:'yellow'});
    el2.toDefs();
    $$('#console2')[0].innerHTML = htmlToTXT(p2.toString());
}
function clearPaper(){
    p2.clear();
    console.log(p2.toString());
    $$('#console2')[0].innerHTML = htmlToTXT(p2.toString());
}
