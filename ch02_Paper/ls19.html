polyline , polygon , path

Paper.polyline() / polygon()

var p = Snap('#paper');
var line = p.polyline(10,80,90,20,90,80,10,20,10,80,90,80).attr({
    strokeWidth: 4,
    fill: 'none'
})
var shape = p.polygon(15,15,85,15,50,40).attr({
    fill: '#2980B9'
})
p.g().add(line).add(shape).attr({
    stroke: '#2980B9',
    strokeLinecap: 'round',
    strokeLinejoin: 'round'
})
