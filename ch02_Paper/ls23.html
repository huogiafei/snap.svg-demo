Paper.mask()

var p2 = Snap('#paper2');
var rg = p2.gradient('r(.5,.5 ,.5)hsla(0,0,255,1):50-hsla(0,0,255,0)');
var mask = p2.mask();
mask.circle(50, 50, 40).attr({
    fill: rg
});
p2.image('../img/pocketman.png', 10, 10, 90, 90).attr({
    mask: mask
});
