Snap.parsePathString() / parseTransformString()

/**
 * parsePathstring(pathStr/arr):解析路径字符串为路径区段数组
 * parseTransformString(tStr/arr)解析给定的变换字符串为变换数组
 */
var arr1 = Snap.parsePathString('M10 10 L 90 90');
var arr2 = Snap.parsePathString(['M10 10 l 90 50', 'm 90 50 l 10 90']);
var arr3 = Snap.parsePathString('t10,10');
var con = document.querySelector('#console');
console.log(arr1, arr2, arr3);
