Snap.hsb() / hsl() / rgb()

var reg = /\d+/g;
var hsbC = 'hsb(158,70%,70%)';
var hslC = 'hsl(158,70%,70%)';
var rgbC = 'rgb(54,179,133)';
var hsbP = hsbC.match(reg);
var hslP = hslC.match(reg);
var rgbP = rgbC.match(reg);
var result = [
    {t: 'hsb()', c: Snap.hsb(hsbP[0],hsbP[1],hsbP[2])},
    {t: 'hsl()', c: Snap.hsl(hslP[0],hslP[1],hslP[2])},
    {t: 'rgb()', c: Snap.rgb(rgbP[0],rgbP[1],rgbP[2])}
]
