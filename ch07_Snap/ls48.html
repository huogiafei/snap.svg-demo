Snap.parse() / fragment()

/**
 * parses : Parses SVG fragment and converts it into a Fragment
 * fragment: Create a DOM fragment from a given list of elements or strings
 */
var p = Snap('#paper');
var fragment = Snap.parse('<circle cx="50" cy="50" r="40"></circle>');
p.add(fragment);
var dom = Snap.fragment(
        '<rect x="100" y="10" width="40" height="80" ></rect>' +
        '<rect x="150" y="10" width="40" height="80" ></rect>'
);
//p.append(dom);
console.log(fragment, dom);
document.querySelector('#paper').appendChild(dom.node);
