var p = Snap('#paper');
var currentMina = '',lock = false;
var circle = p.circle(100, 25, 20).attr({
    fill: '#088'
});
function ani() {
    var cx = circle.attr('cx');
    circle.animate({
        cx: (cx == 100 ? 700 : 100),
    }, 2000, mina[currentMina], function () {
        $$(".btn-box button").forEach(function (btn) {
            btn.style.opacity = 1
        })
        lock = false;
    });
}
function mode(arg) {
    if (!lock) {
        lock = true;
        switch (arg) {
            case 1: currentMina = 'easein'; break;
            case 2: currentMina = 'easeinout'; break;
            case 3: currentMina = 'easeout'; break;
            case 4: currentMina = 'linear'; break;
        }
        $$(".btn-box button").forEach(function (btn, i) {
            if (arg !== (i + 1)) {
                btn.style.opacity = 0.5
            }
        })
        ani();
    }
}
